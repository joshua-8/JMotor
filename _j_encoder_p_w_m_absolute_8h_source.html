190  void run()
191  {
192  if (newAngle) {
193  long interval = dataEndMicros - dataStartMicros;
194  long cycle = dataEndMicros - oldDataEndMicros;
195  if (interval > 0 && cycle > 0) { //check that interrupts didn't change one number but not the other
196  angle = constrain(interval * ps.PWM_STEPS / cycle - ps.MIN_STEPS, 0, ps.RESOLUTION);
197  newAngle = false;
198  if (abs(angle - lastAngle) > ps.RESOLUTION / 2) { // angle jump over half of circle is assummed to be the shorter crossing of 0
199  if (angle > lastAngle) {
200  turns--;
201  } else {
202  turns++;
203  }
204  }
205  }
206  }
207  long velDist = (angle - lastVelAngle) + (turns - lastVelTurns) * ps.RESOLUTION;
208  if (micros() - lastVelTimeMicros > velEnoughTime || abs(velDist) > velEnoughTicks) {
209  velocity = (double)1000000.0 * velDist / (micros() - lastVelTimeMicros) * distPerCountFactor * reverse;
210  lastVelTimeMicros = micros();
211  newSpeed = true;
212  lastVelAngle = angle;
213  lastVelTurns = turns;
214  } else {
215  newSpeed = false;
216  }
