194  {

196  long interval = dataEndMicros - dataStartMicros;
197  long cycle = dataEndMicros - oldDataEndMicros;
198  if (interval > 0 && cycle > 0) { //check that interrupts didn't change one number but not the other
199  angle = constrain(interval * ps.PWM_STEPS / cycle - ps.MIN_STEPS, 0, ps.RESOLUTION);
200  newAngle = false;
201  if (abs(angle - lastAngle) > ps.RESOLUTION / 2) { // angle jump over half of circle is assummed to be the shorter crossing of 0
202  if (angle > lastAngle) {
203  turns--;
204  } else {
205  turns++;
206  }
207  }
208  }
209  }
210  long velDist = (angle - lastVelAngle) + (turns - lastVelTurns) * ps.RESOLUTION;
211  if (micros() - lastVelTimeMicros > velEnoughTime || abs(velDist) > velEnoughTicks) {
212  velocity = (double)1000000.0 * velDist / (micros() - lastVelTimeMicros) * distPerCountFactor * reverse;
