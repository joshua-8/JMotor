589 writeAngleToServo(float ang)
590 {
591 ang = constrain(ang, min(minAngleLimit, maxAngleLimit), max(minAngleLimit, maxAngleLimit));
592 if ((reverse) != (minAngleLimit < maxAngleLimit)) {
593 ang = floatMap(ang, maxSetAngle, minSetAngle, servo.getMinRange(), servo.getMaxRange());
594 } else { // not reversed
595 ang = floatMap(ang, minSetAngle, maxSetAngle, servo.getMinRange(), servo.getMaxRange());
596 }
597 if (enabled && !sleeping) {
598 servo.set(ang);
599 }
600 }
601 float floatMap(float x, float in_min, float in_max, float out_min, float out_max)
602 {
603 return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
604 }
