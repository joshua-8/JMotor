170  void run()
171  {
172  angle = readAngle();
173  if (abs(angle - lastAngle) > STEPS_PER_TURN / 2) { // angle jump over half of circle is assummed to be the shorter crossing of 0
174  if (angle > lastAngle) {
175  turns--;
176  } else {
177  turns++;
178  }
179  }
180 
181  long velDist = (angle - lastVelAngle) + (turns - lastVelTurns) * STEPS_PER_TURN;
182  if (micros() - lastVelTimeMicros > velEnoughTime || abs(velDist) > velEnoughTicks) {
183  velocity = (double)1000000.0 * velDist / (micros() - lastVelTimeMicros) * distPerCountFactor * reverse;
184  lastVelTimeMicros = micros();
185  newSpeed = true;
186  lastVelAngle = angle;
187  lastVelTurns = turns;
188  } else {
189  newSpeed = false;
190  }
191 
192  lastAngle = angle;
193  }
