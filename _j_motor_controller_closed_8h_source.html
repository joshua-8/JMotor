83 void run()
84 {
85 encoder.run();
86 time = (micros() - lastRunMicros) / 1000000.0;
87 if (time == 0)
88 return;
89 lastRunMicros = micros();
90 if (getEnable()) {
91 if (open) {
92 if (velSetpoint != velSetpointTarget) {
93 if (posSetpointSmoother.getPreventGoingWrongWay() && velSetpointTarget != 0 && velSetpoint != 0 && ((velSetpointTarget > 0) != (velSetpoint > 0))) {
94 velSetpoint = 0;
95 }
96 velSetpoint += constrain(velSetpointTarget - velSetpoint, -accelLimit * time, accelLimit * time);
97 }
98 velSetpoint = constrain(velSetpoint, -compensator.getMaxVel() * (getDriverMinRange() < 0), compensator.getMaxVel() * (getDriverMaxRange() > 0));
99 velSetpoint = constrain(velSetpoint, -velLimit, velLimit);
100 setVal = compensator.compensate(velSetpoint);
101
102 if (!encoder.hasDirection()) {
103 encoder.setRev((setVal <= 0));
104 }
105
106 driverInRange = driver.set(setVal);
107 posSetpoint = encoder.getPos();
108 posDeltaSetpoint = velSetpoint;
109 } else { //closed loop
110 if (posMode) {
111 if (smoothed) { //setPosTarget
112 posSetpointSmoother.setPosition(posSetpoint);
113 posSetpoint = posSetpointSmoother.calc();
114 posDeltaSetpoint = posSetpointSmoother.getVelocity();
115 if (limitSetpointDistFromCurrent) {
116 posSetpoint = constrain(posSetpoint, encoder.getPos() - distFromSetpointLimit, encoder.getPos() + distFromSetpointLimit);
117 posSetpointSmoother.setPosition(posSetpoint);
118 }
119 } else { //setPosSetpoint
120 posDeltaSetpoint = (posSetpoint - lastPosSetpoint) / time;
121 //nothing to do here, posSetpoint was set
122 }
123 } else { //posDelta
124 if (posDeltaSetpoint != posDeltaSetpointTarget) {
125 if (posSetpointSmoother.getPreventGoingWrongWay() && posDeltaSetpointTarget != 0 && posDeltaSetpoint != 0 && ((posDeltaSetpointTarget > 0) != (posDeltaSetpoint > 0))) {
126 posDeltaSetpoint = 0; //prevent going wrong way
127 }
128 posDeltaSetpointTarget = constrain(posDeltaSetpointTarget, -velLimit, velLimit);
129 posDeltaSetpointTarget = constrain(posDeltaSetpointTarget, -compensator.getMaxVel() * (getDriverMinRange() < 0), compensator.getMaxVel() * (getDriverMaxRange() > 0));
130 posDeltaSetpoint += constrain(posDeltaSetpointTarget - posDeltaSetpoint, -accelLimit * time, accelLimit * time);
131 }
132 posSetpoint += posDeltaSetpoint * time;
133 if (limitSetpointDistFromCurrent)
134 posSetpoint = constrain(posSetpoint, encoder.getPos() - distFromSetpointLimit, encoder.getPos() + distFromSetpointLimit);
135 }
136 //MAKE MOTOR GO TO posSetpoint, by setting velSetpointTarget
137
138 velSetpointTarget = controlLoop.calc(this);
139
140 lastPosSetpoint = posSetpoint;
141
142 velSetpoint = velSetpointTarget; //open loop uses velSetpointTarget and velsetpoint for acceleration, but closed loop has posDeltaSetpointTarget and posDeltaSetpoint, so just set them equal
143
144 setVal = compensator.compensate(velSetpoint);
145
146 if (!encoder.hasDirection()) {
147 encoder.setRev((setVal < 0));
148 }
149
150 driverInRange = driver.set(setVal);
151 } //end of closed loop mode
152 } //enabled
