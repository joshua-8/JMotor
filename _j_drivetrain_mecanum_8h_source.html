200 {
201 m.FRmotor = t.x * robotToWheelScalar.x + t.y * robotToWheelScalar.y + t.theta * robotToWheelScalar.theta;
202 m.FLmotor = t.x * robotToWheelScalar.x - t.y * robotToWheelScalar.y - t.theta * robotToWheelScalar.theta;
203 m.BLmotor = t.x * robotToWheelScalar.x + t.y * robotToWheelScalar.y - t.theta * robotToWheelScalar.theta;
